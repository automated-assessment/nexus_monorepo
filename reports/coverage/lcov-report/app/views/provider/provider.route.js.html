(function(){
    angular
        .module('PeerFeedback')
        .config(['$stateProvider',function($stateProvider){
            $stateProvider
                .state('frameState.providerState',{
                    url:'/provider?receiverSid?providerSid',
                    params:{
                        receiverSid:null,
                        providerSid:null,
                        aid:null
                    },
                    templateUrl:'app/views/provider/provider.html',
                    controller:'ProviderController as vm',
                    resolve:{
 
                        provider:['$stateParams','allocationService',function($stateParams,allocationService){
                            return allocationService.getOneAllocation($stateParams.receiverSid,$stateParams.providerSid)
                                .then(function(response){
                                    return response.data;
                                });
                        }],
                        snippets:['$stateParams','gitSubmission',function($stateParams,gitSubmission){
                            return gitSubmission.getSnippets($stateParams.receiverSid,$stateParams.providerSid,$stateParams.aid);
                                // .then(function(snippets){
                                //     vm.snippets = snippets;
                                // });
                        }]
 
 
 
 
 
 
 
 
 
 
 
 
                    }
                });
 
        }]);
}());
